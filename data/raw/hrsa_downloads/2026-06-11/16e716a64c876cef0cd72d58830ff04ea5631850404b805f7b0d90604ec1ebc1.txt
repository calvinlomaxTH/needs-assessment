	<script>
		$(document).ready(function () {
			const items = $('.usa-card-group .usa-card'); // updated selector
			const totalItems = items.length;
			const dropdown = $('#PageSizeBottom');
			let itemsPerPage = 10;
			let totalPages = 1;

			// function that scrolls to form HTML element, when filters are selected
			function focusDiv() {
                const firstDiv = document.getElementById("frmDataDl");
				if (firstDiv) {
                    firstDiv.scrollIntoView({ behavior: 'smooth' });
				}
			}

			// checks if the form was submitted to the server/POST call made

			totalPages = Math.ceil(totalItems / itemsPerPage);

			function showPage(page) {
				const start = (page - 1) * itemsPerPage;
				const end = Math.min(start + itemsPerPage, totalItems);
				items.hide().slice(start, end).show();

				updatePaginationMessage(start + 1, end, totalItems);
				renderPageNumbers(page);
				toggleNavButtons(page);
			}

			function updatePaginationMessage(start, end, total) {
				const message = `Showing
										  <span class="text-bold margin-x-1">${start} - ${end}</span> of
										  <span class="text-bold margin-x-1">${total}</span>
										  <span class="tablet:border-right-1px tablet:border-base tablet:padding-right-2 tablet:margin-right-2">data sets</span>`;
				$('.result-number').html(message);

				const currentPageCount = end - start + 1;
				const msg = `<span class="font-ui-xs">Viewing ${currentPageCount} of ${total} data sets</span>`;
				$('.dataset-count').html(msg);
			}

			function renderPageNumbers(currentPage) {
				const maxVisible = 5;
				const container = $('.usa-pagination__list');
				container.find('.usa-pagination__page-no').remove();

				let startPage = Math.max(1, currentPage - Math.floor(maxVisible / 2));
				let endPage = Math.min(totalPages, startPage + maxVisible - 1);

				if (endPage - startPage < maxVisible - 1) {
					startPage = Math.max(1, endPage - maxVisible + 1);
				}

				const nextButton = container.find('.next').closest('li');

				for (let i = startPage; i <= endPage; i++) {
					const isActive = i === currentPage ? 'usa-current' : '';
					const aria = i === currentPage ? 'aria-current="page"' : '';
					const li = $(`
						  <li class="usa-pagination__item usa-pagination__page-no">
							  <a href="#" class="usa-pagination__button ${isActive}" ${aria} data-page="${i}">${i}</a>
						  </li>`);

					li.insertBefore(nextButton);
				}
			}

			function toggleNavButtons(currentPage) {
				$('.first, .prev').toggleClass('usa-disabled', currentPage === 1);
				$('.last, .next').toggleClass('usa-disabled', currentPage === totalPages);
			}

			dropdown.change(function () {
				itemsPerPage = parseInt($(this).val());
				totalPages = Math.ceil(totalItems / itemsPerPage);
				showPage(1);
			});

			$(document).on('click', '.usa-pagination__button[data-page]', function (e) {
				e.preventDefault();
				const page = parseInt($(this).data('page'));
				showPage(page);
			});

			$('.first').click(() => showPage(1));
			$('.last').click(() => showPage(totalPages));
			$('.prev').click(() => {
				const current = parseInt($('.usa-pagination__button.usa-current').data('page'));
				if (current > 1) showPage(current - 1);
			});
			$('.next').click(() => {
				const current = parseInt($('.usa-pagination__button.usa-current').data('page'));
				if (current < totalPages) showPage(current + 1);
			});

			showPage(1);
		});
    </script>
